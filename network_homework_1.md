# Задание 1. Перевод чисел

## 137

### Перевод в 2СС через деление на 2
137 / 2 = 68 остаток 1
68 / 2 = 34 остаток 0
34 / 2 = 17 остаток 0
17 / 2 = 8 остаток 1
8 / 2 = 4 остаток 0
4 / 2 = 2 остаток 0
2 / 2 = 1 остаток 0
1 / 2 = 0 остаток 1

Итог:
137(10) = 10001001(2)

### Перевод в 8СС
10001001->010 001 001
010 = 2
001 = 1
001 = 1

Итог:
137(10) = 211(8)

### Перевод в 16СС
10001001->1000 1001
1000 = 8
1001 = 9

Итог:
137(10) = 89(16)

## 245

### Перевод в 2СС через деление на 2
245 / 2 = 122 остаток 1
122 / 2 = 61 остаток 0
61 / 2 = 30 остаток 1
30 / 2 = 15 остаток 0
15 / 2 = 7 остаток 1
7 / 2 = 3 остаток 1
3 / 2 = 1 остаток 1
1 / 2 = 0 остаток 1

Итог:
245(10) = 11110101(2)

### Перевод в 8СС
11110101->011 110 101
011 = 3
110 = 6
101 = 5

Итог:
245(10) = 365(8)

### Перевод в 16СС
11110101->1111 0101
1111 = F
0101 = 5

Итог:
245(10) = F5(16)
